根据syri输出，提取共线性位置

```python
import re
import sys

# END=110498;ChrB=Chr05;StartB=41613;EndB=175279
with open (sys.argv[1]) as f:
    for i in f:
        if i.startswith('#'):           
            pass
        else:
            i = i.strip().split('\t')
            info = i[7]
            end = re.findall('END=(.+?);',info)
            s_chr = re.findall('ChrB=(.+?);',info)
            s_start = re.findall('StartB=(.+?);',info)
            s_end = re.findall('EndB=(.+?);',info)
            
            end = ''.join(end)
            s_chr = ''.join(s_chr)
            s_start = ''.join(s_start)
            s_end = ''.join(s_end)        
            
            # 去掉左右的<>符号
            region_type = re.findall('<(.+?)>',i[4])
            region_type = ''.join(region_type)
            
            out = '{}\t{}\t{}\t{}\t{}\t{}\t{}'.format(i[0], i[1], end, region_type, s_chr, s_start, s_end)
            print(out)
```
